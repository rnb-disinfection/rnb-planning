82  void recurse_set_param(ros::NodeHandlePtr _node_handle, string key_prev, YAML::Node::iterator it_doc, YAML::Node::iterator it_end){
83  if (it_doc==it_end){
84  return;
85  }
86  stringstream stream1;
87  if (key_prev.size()==0) {
88  stream1<< it_doc->first;
89  }
90  else{
91  stream1<<key_prev<<"/"<< it_doc->first;
92  }
93  string current_key = stream1.str();
94  if(it_doc->second.IsMap()) {
95  recurse_set_param(_node_handle, current_key, it_doc->second.begin(), it_doc->second.end());
96  }
97  else if(it_doc->second.IsDefined()){
98  set_rosparam_from_yaml_node(_node_handle, current_key, it_doc->second);
99  }
100  else{
101  PRINT_ERROR("ERROR::NON_DEFINED NODE" + current_key);
102  exit(0);
103  }
104  recurse_set_param(_node_handle, key_prev, ++it_doc, it_end);
105  }
